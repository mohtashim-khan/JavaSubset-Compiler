boolean globalBoolTrue;
boolean globalBoolFalse;

main()
{
    boolean T;
    boolean F;

    boolean resultOR;
    boolean resultAND;

    boolean resultGlobal;

    T = true;
    F = false;

    globalBoolFalse = false;
    globalBoolTrue = true;

    prints("expecting true: \n");
    printb(T);
    prints("\n");


    prints("expecting false: \n");
    printb(F);
    prints("\n");


    resultOR = T || F;
    resultAND = T && F;

    resultGlobal = globalBoolTrue && globalBoolFalse;

    prints("expecting true resultOR: \n");
    printb(resultOR);
    prints("\n");

    prints("expecting false resultAND: \n");
    printb(resultAND);
    prints("\n");

    prints("expecting true orfunc: \n");
    printb(orfunc(T,F));
    prints("\n");

    prints("expecting false andfunc: \n");
    printb(andfunc(T,F));
    prints("\n");

    prints("expecting false resultGlobal: \n");
    printb(resultGlobal);
    prints("\n");
}

boolean andfunc(boolean a, boolean b)
{
        return a && b;
}

boolean orfunc(boolean a, boolean b)
{
        return a || b;
}
